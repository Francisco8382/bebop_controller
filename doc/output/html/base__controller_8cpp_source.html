89 void BaseController::Odometry_CB(const geometry_msgs::PoseStamped& pose_msg) {
90 ROS_INFO_ONCE("Controller got first pose message.");
91 timeOut.stop();
92 timeOut.start();
93 if (waypointHasBeenPublished_) {
94 if (!takeoff) {
95 TakeOff();
96 }
97 if (!controller_active_) {
98 lastTime = ros::Time::now();
99 }
100 const Eigen::Vector3d position = Eigen::Vector3d(pose_msg.pose.position.x,
101 pose_msg.pose.position.y,
102 pose_msg.pose.position.z);
103 const Eigen::Quaterniond orientation = Eigen::Quaterniond(pose_msg.pose.orientation.w,
104 pose_msg.pose.orientation.x,
105 pose_msg.pose.orientation.y,
106 pose_msg.pose.orientation.z);
107 const Eigen::Vector3d zeros = Eigen::Vector3d(0,0,0);
108 mav_msgs::EigenOdometry odometry = mav_msgs::EigenOdometry(position,orientation,zeros,zeros);
109 ros::Time now = ros::Time::now();
110 diff = now.toSec() - lastTime.toSec();
111 lastTime = now;
112 SetOdometry(odometry);
113 EstimateVelocity();
114 EstimateAcceleration();
115 if (!CheckSafeZone()) {
116 Stop(true);
117 }
118 geometry_msgs::Twist ref_command_signals;
119 CalculateCommandVelocities(ref_command_signals);
120 if (!disable_commands && !stop) {
121 motor_velocity_reference_pub_.publish(ref_command_signals);
122 }
123 }
124 }

169 void BaseController::GetErrors(Vector4& e) {
170 e.x = state.position.x - command_trajectory_.position_W[0];
171 e.y = state.position.y - command_trajectory_.position_W[1];
172 e.z = state.position.z - command_trajectory_.position_W[2];
173 e.yaw = state.orientation.z - yawFromQuaternion(command_trajectory_.orientation_W_B);
174 while (std::abs(e.yaw) > M_PI) {
175 if (e.yaw > 0) {
176 e.yaw -= 2*M_PI;
177 }
178 else {
179 e.yaw += 2*M_PI;
180 }
181 }
182 }

214 void BaseController::EstimateVelocity() {
215 state.velocity.x = (state.position.x - last_state.position.x)/diff;
216 state.velocity.y = (state.position.y - last_state.position.y)/diff;
217 state.velocity.z = (state.position.z - last_state.position.z)/diff;
218 state.angular_velocity.x = (state.orientation.x - last_state.orientation.x)/diff;
219 state.angular_velocity.y = (state.orientation.y - last_state.orientation.y)/diff;
220 state.angular_velocity.z = (state.orientation.z - last_state.orientation.z)/diff;
221 last_state.position = state.position;
222 last_state.orientation = state.orientation;
223 }

226 void BaseController::EstimateAcceleration() {
227 state.acceleration.x = (state.velocity.x - last_state.velocity.x)/diff;
228 state.acceleration.y = (state.velocity.y - last_state.velocity.y)/diff;
229 state.acceleration.z = (state.velocity.z - last_state.velocity.z)/diff;
230 state.angular_acceleration.x = (state.angular_velocity.x - last_state.angular_velocity.x)/diff;
231 state.angular_acceleration.y = (state.angular_velocity.y - last_state.angular_velocity.y)/diff;
232 state.angular_acceleration.z = (state.angular_velocity.z - last_state.angular_velocity.z)/diff;
233 last_state.velocity = state.velocity;
234 last_state.angular_velocity = state.angular_velocity;

264 void BaseController::CalculateLeashLength(Vector4& e, Vector4& P) {
265 Vector3 Ppos, vel, accel;
266 Ppos.x = P.x*sqrt(std::abs(e.x));
267 Ppos.y = P.y*sqrt(std::abs(e.y));
268 vel.x = std::abs(max(state.velocity.x,MIN_VEL));
269 vel.y = std::abs(max(state.velocity.y,MIN_VEL));
270 accel.x = std::abs(max(state.acceleration.x,MIN_ACCEL));
271 accel.y = std::abs(max(state.acceleration.y,MIN_ACCEL));
272 leash_length.x = accel.x/(2*pow(Ppos.x,2)) + pow(vel.x,2)/(2*accel.x);
273 leash_length.y = accel.y/(2*pow(Ppos.y,2)) + pow(vel.y,2)/(2*accel.y);
274 }
