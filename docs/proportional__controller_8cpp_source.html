60 void ProportionalController::CalculateCommandVelocities(geometry_msgs::Twist& ref_command_signals){
61 if (!controller_active_){
62 ref_command_signals.linear.x = 0.0;
63 ref_command_signals.linear.y = 0.0;
64 ref_command_signals.linear.z = 0.0;
65 ref_command_signals.angular.z = 0.0;
66 return;
67 }
68 Vector4 u, e;
69 GetErrors(e);
70 CalculateLeashLength(e, P);
71 LimitPositionErrors(e);
72
73 u.x = (-P.x*e.x + RG.x*command_trajectory_.velocity_W[0])/norm.horizontal;
74 u.y = (-P.y*e.y + RG.y*command_trajectory_.velocity_W[1])/norm.horizontal;
75 u.z = (-P.z*e.z + RG.z*command_trajectory_.velocity_W[2])/norm.vertical;
76 u.yaw = (-P.yaw*e.yaw + RG.yaw*command_trajectory_.angular_velocity_W[2])*RAD_2_DEG/norm.rotation;
77
78 ref_command_signals.linear.x = clamp(cos(state.orientation.z)*u.x - sin(state.orientation.z)*u.y,max_speed.x);
79 ref_command_signals.linear.y = clamp(sin(state.orientation.z)*u.x + cos(state.orientation.z)*u.y,max_speed.y);
80 ref_command_signals.linear.z = clamp(u.z,max_speed.z);
81 ref_command_signals.angular.z = clamp(u.yaw,max_speed.yaw);
82 }
