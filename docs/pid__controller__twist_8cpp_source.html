94 void PIDController::CalculateCommandVelocities(geometry_msgs::Twist& ref_command_signals){
95 if (!controller_active_){
96 ref_command_signals.linear.x = 0.0;
97 ref_command_signals.linear.y = 0.0;
98 ref_command_signals.linear.z = 0.0;
99 ref_command_signals.angular.z = 0.0;
100 return;
101 }
102 Vector4 a, e, dot_e;
103 GetErrors(e);
104 CalculateLeashLength(e, P);
105 LimitPositionErrors(e);
106 IntegrateErrors(e);
107 LimitIntegralPart();
108 GetVelocityErrors(dot_e);
109
110 a.x = -P.x*e.x - D.x*dot_e.x - I.x*int_e.x + RG.z*command_trajectory_.acceleration_W[0];
111 a.y = -P.y*e.y - D.y*dot_e.y - I.y*int_e.y + RG.z*command_trajectory_.acceleration_W[1];
112 a.z = -P.z*e.z - D.z*dot_e.z - I.z*int_e.z + RG.z*command_trajectory_.acceleration_W[2];
113 a.yaw = -P.yaw*e.yaw - D.yaw*dot_e.yaw - I.yaw*int_e.yaw;
114
115 u.x = clamp(u.x + (diff*a.x)/norm.horizontal,max_speed.x);
116 u.y = clamp(u.y + (diff*a.y)/norm.horizontal,max_speed.y);
117 u.z = clamp(u.z + (diff*a.z)/norm.vertical,max_speed.z);
118 u.yaw = clamp(u.yaw + (diff*a.yaw)*RAD_2_DEG/norm.rotation,max_speed.yaw);
119
120 ref_command_signals.linear.x = clamp(cos(state.orientation.z)*u.x - sin(state.orientation.z)*u.y,max_speed.x);
121 ref_command_signals.linear.y = clamp(sin(state.orientation.z)*u.x + cos(state.orientation.z)*u.y,max_speed.y);
122 ref_command_signals.linear.z = clamp(u.z,max_speed.z);
123 ref_command_signals.angular.z = clamp(u.yaw,max_speed.yaw);
124 }
