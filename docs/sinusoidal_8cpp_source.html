34 void Sinusoidal::Trajectory_CB(const ros::TimerEvent& event) {
35 ros::Time now = ros::Time::now();
36 double x, y, z, yaw;
37 double dot_x, dot_y, dot_z, dot_yaw;
38 double ddot_x, ddot_y, ddot_z;
39 double t = now.toSec() - Initial_Time.toSec() + wp_params_.DiffTime;
40 double W1, W2, W3;
41 W1 = W3 = 2*M_PI/wp_params_.TrajectoryTime;
42 W2 = 4*M_PI/wp_params_.TrajectoryTime;
43
44 x = t_params_.TrajectoryDistance.x()*sin(W1*t) + t_params_.PositionBeforeTrajectory.x();
45 y = t_params_.TrajectoryDistance.y()*sin(W2*t) + t_params_.PositionBeforeTrajectory.y();
46 z = t_params_.TrajectoryDistance.z()*sin(W3*t) + t_params_.PositionBeforeTrajectory.z();
47
48 dot_x = t_params_.TrajectoryDistance.x()*W1*cos(W1*t);
49 dot_y = t_params_.TrajectoryDistance.y()*W2*cos(W2*t);
50 dot_z = t_params_.TrajectoryDistance.z()*W3*cos(W3*t);
51
52 ddot_x = -t_params_.TrajectoryDistance.x()*pow(W1,2)*sin(W1*t);
53 ddot_y = -t_params_.TrajectoryDistance.y()*pow(W2,2)*sin(W2*t);
54 ddot_z = -t_params_.TrajectoryDistance.z()*pow(W3,2)*sin(W3*t);
55
56 switch (status)
57 {
58 case BeforeTrajectory:
59 case AfterTrajectory:
60 x = t_params_.PositionBeforeTrajectory.x();
61 y = t_params_.PositionBeforeTrajectory.y();
62 z = t_params_.PositionBeforeTrajectory.z();
63 if (t_params_.Yaw_Enabled){
64 yaw = atan2(2*t_params_.TrajectoryDistance.y(),t_params_.TrajectoryDistance.x()) + t_params_.Yaw_Offset;
65 }
66 else {
67 yaw = 0;
68 }
69 dot_x = dot_y = dot_z = ddot_x = ddot_y = ddot_z = dot_yaw = 0;
70 break;
71
72 case Trajectory:
73 if (t > wp_params_.TrajectoryTime) {
74 status = AfterTrajectory;
75 return;
76 }
77 if (t_params_.Yaw_Enabled){
78 yaw = atan2(dot_y,dot_x) + t_params_.Yaw_Offset;
79 dot_yaw = ((dot_x*ddot_y - dot_y*ddot_x)/pow(dot_x,2))/(1+pow(dot_y/dot_x,2));
80 }
81 else {
82 yaw = dot_yaw = 0;
83 }
84 break;
85 }
86
87 Eigen::Vector3d desired_position(x, y, z);
88 Eigen::Vector3d velocities(dot_x,dot_y,dot_z);
89 Eigen::Vector3d accelerations(ddot_x,ddot_y,ddot_z);
90
91 mav_msgs::EigenTrajectoryPoint point;
92 point.position_W = desired_position;
93 point.setFromYaw(yaw);
94 point.velocity_W = velocities;
95 point.setFromYawRate(dot_yaw);
96 point.acceleration_W = accelerations;
97
98 mav_msgs::msgMultiDofJointTrajectoryFromEigen(point, &position_target_);
99
100 setpoint_pub_.publish(position_target_);
101 }
