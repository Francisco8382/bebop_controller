98 void CITCController::CalculateCommandVelocities(geometry_msgs::Twist& ref_command_signals){
99 if (!controller_active_){
100 ref_command_signals.linear.x = 0.0;
101 ref_command_signals.linear.y = 0.0;
102 ref_command_signals.linear.z = 0.0;
103 ref_command_signals.angular.z = 0.0;
104 return;
105 }
106 Vector4 a, v_nom, v_stc, e, dot_e, ddot_e;
107 GetErrors(e);
108 GetVelocityErrors(dot_e);
109 GetAccelerationErrors(ddot_e, dot_e);
110
111 v_nom.x = -K1.x*pow(std::abs(dot_e.x),sigma)*sgn(dot_e.x) - K2.x*pow(std::abs(e.x),((sigma)/(2-sigma)))*sgn(e.x);
112 v_nom.y = -K1.y*pow(std::abs(dot_e.y),sigma)*sgn(dot_e.y) - K2.y*pow(std::abs(e.y),((sigma)/(2-sigma)))*sgn(e.y);
113 v_nom.z = -K1.z*pow(std::abs(dot_e.z),sigma)*sgn(dot_e.z) - K2.z *pow(std::abs(e.z),((sigma)/(2-sigma)))*sgn(e.z);
114 v_nom.yaw = -K1.yaw*pow(std::abs(dot_e.yaw),sigma)*sgn(dot_e.yaw) - K2.yaw *pow(std::abs(e.yaw),((sigma)/(2-sigma)))*sgn(e.yaw);
115
116 IntegrateZetaAndEta(v_nom, ddot_e);
117
118 v_stc.x = -K3.x*std::abs(int_zeta.x)*sgn(int_zeta.x) + int_eta.x;
119 v_stc.y = -K3.y*std::abs(int_zeta.y)*sgn(int_zeta.y) + int_eta.y;
120 v_stc.z = -K3.z *std::abs(int_zeta.z)*sgn(int_zeta.z) + int_eta.z;
121 v_stc.yaw = -K3.yaw *std::abs(int_zeta.yaw)*sgn(int_zeta.yaw) + int_eta.yaw;
122
123 a.x = clamp(v_nom.x + v_stc.x + RG.x*command_trajectory_.acceleration_W[0],
124 mass*(lambda.x-std::abs(command_trajectory_.acceleration_W[0])));
125 a.y = clamp(v_nom.y + v_stc.y + RG.y*command_trajectory_.acceleration_W[1],
126 mass*(lambda.y-std::abs(command_trajectory_.acceleration_W[1])));
127 a.z = clamp(v_nom.z + v_stc.z + RG.z*command_trajectory_.acceleration_W[2],
128 mass*(lambda.z-std::abs(command_trajectory_.acceleration_W[2])));
129 a.yaw = v_nom.yaw + v_stc.yaw;
130
131 double u_Terr, u_T, phi_r, theta_r, psi;
132 u_Terr = a.z + mass*GRAVITY;
133 u_T = sqrt(pow(a.x,2) + pow(a.y,2) + pow(u_Terr,2));
134 psi = state.orientation.z;
135 theta_r = atan(((a.x * cos(psi)) + (a.y * sin(psi)))/u_Terr);
136 phi_r = atan(cos(theta_r)*(((a.x * sin(psi)) - (a.y * cos(psi)))/(u_Terr)));
137
138 u_z = clamp(u_z + diff*a.z,max_speed.z*norm.vertical);
139
140 ref_command_signals.linear.x = clamp(theta_r*RAD_2_DEG/norm.angle,max_speed.x);
141 ref_command_signals.linear.y = clamp(-phi_r*RAD_2_DEG/norm.angle,max_speed.y);
142 ref_command_signals.linear.z = clamp(u_z/norm.vertical,max_speed.z);
143 ref_command_signals.angular.z = clamp(a.yaw*RAD_2_DEG/norm.rotation,max_speed.yaw);
144 }

146 void CITCController::GetAccelerationErrors(Vector4& ddot_e, Vector4& dot_e) {
147 ddot_e.x = (dot_e.x - last_dot_e.x)/diff;
148 ddot_e.y = (dot_e.y - last_dot_e.y)/diff;
149 ddot_e.z = (dot_e.z - last_dot_e.z)/diff;
150 ddot_e.yaw = (dot_e.yaw - last_dot_e.yaw)/diff;
151 last_dot_e.x = dot_e.x;
152 last_dot_e.y = dot_e.y;
153 last_dot_e.z = dot_e.z;
154 last_dot_e.yaw = dot_e.yaw;

157 void CITCController::IntegrateZetaAndEta(Vector4& v_nom, Vector4& ddot_e) {
158 Vector4 zeta, eta;
159 zeta.x = ddot_e.x - v_nom.x;
160 zeta.y = ddot_e.y - v_nom.y;
161 zeta.z = ddot_e.z - v_nom.z;
162 zeta.yaw = ddot_e.yaw - v_nom.yaw;
163 int_zeta.x += zeta.x*diff;
164 int_zeta.y += zeta.y*diff;
165 int_zeta.z += zeta.z*diff;
166 int_zeta.yaw += zeta.yaw*diff;
167 eta.x = -K4.x*sgn(int_zeta.x);
168 eta.y = -K4.y*sgn(int_zeta.y);
169 eta.z = -K4.z*sgn(int_zeta.z);
170 eta.yaw = -K4.yaw*sgn(int_zeta.yaw);
171 int_eta.x += eta.x*diff;
172 int_eta.y += eta.y*diff;
173 int_eta.z += eta.z*diff;
174 int_eta.yaw += eta.yaw*diff;
175 }
